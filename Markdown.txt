# DevOps Internship – Task 4
# Git Commands Log

# 1. Initialize Local Repository
git init

# 2. Add Files to Staging
git add .

# 3. Commit Initial Code
git commit -m "Initial commit"

# 4. Link Local Repo to GitHub
git remote add origin <your-repo-url>

# 5. Push to Main Branch
git branch -M main
git push -u origin main

# 6. Create Development Branch
git branch dev
git checkout dev
git push -u origin dev

# 7. Create Feature Branch
git branch feature
git checkout feature
git push -u origin feature

# 8. Switch Between Branches
git checkout dev
git checkout feature

# 9. Merge Feature into Dev
git checkout dev
git merge feature

# 10. Push Merged Changes
git push origin dev

# 11. Create Pull Request (via GitHub UI)
# PR: feature → dev, then dev → main

# 12. Add .gitignore
echo "node_modules/" >> .gitignore
echo ".env" >> .gitignore
echo "*.log" >> .gitignore
git add .gitignore
git commit -m "Added .gitignore"
git push

# 13. Create Tag for Release
git tag -a v1.0 -m "First stable release"
git push origin v1.0

# 14. Use Git Stash (Example)
git stash
git stash pop

# 15. View Commit History
git log --oneline --graph --all
